# Show More

Styling component to show expanded or collapsed content.

<Playground
  title="General"
  desc="Basic Usage"
  scope={{ ShowMore, useState , Skeleton}}
  code={`
() => {
  const [expanded, setExpanded] = useState(false);
  return (
    <ShowMore expanded={expanded} onClick={() => setExpanded(!expanded)}>
      <Skeleton w={10} h={20} />
    </ShowMore>
  );
}
`}
/>

<Playground
  title="Customized"
  desc="Personalized title and Show/Hide lines."
  scope={{ ShowMore, useState, Skeleton }}
  code={`
() => {
  const [expanded, setExpanded] = useState(false);
  return (
    <div>
      <ShowMore 
        expanded={expanded} 
        onClick={() => setExpanded(!expanded)} 
        showLines={false} 
        showMoreTitle="Expand" 
        showLessTitle="Collapse"
      >
        <Skeleton w={10} h={20} mt={1}/>
      </ShowMore>
    </div>
  );
}
`}
/>
