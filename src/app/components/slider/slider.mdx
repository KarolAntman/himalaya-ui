# Slider

Display the current value and an inputable range.

<Playground
  desc="Accept clicks or drag actions."
  scope={{ Slider }}
  code={`
  <Slider initialValue={20} />
`}
/>

<Playground
  title="Event"
  desc="Get event when slide."
  scope={{ Slider, useState }}
  code={`
() => {
  const [value, setValue] = useState(20)
  const handler = val => {
    console.log(val)
    setValue(val)
  }
  return <Slider value={value} onChange={handler} w="75%" />
}
`}
/>

<Playground
  title="Range"
  desc="Set the `value` or `initialValue` prop to an array of two numbers, and the slider becomes a range slider."
  scope={{ Slider, useState }}
  code={`
() => {
  const [value, setValue] = useState([20 , 80])
  const handler = val => {
    console.log(val)
    setValue(val)
  }
  return <Slider value={value} onChange={handler} w="75%" />
}
`}
/>
