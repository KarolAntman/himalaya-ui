# Mobile Menu

Create mobile-friendly navigation menus with ease using the `MobileMenu` UI component. 

<Playground
  scope={{ MobileMenu, MobileMenuButton, MobileMenuProvider, SwipeProvider }}
  title={'Mobile Menu example'}
  desc={'Example of a mobile menu'}
  code={`
  () => {
     return (
    <SwipeProvider>
      <MobileMenuProvider>
        <MobileMenuButton notHiding={true}></MobileMenuButton>
        <MobileMenu>
          <MobileMenuButton></MobileMenuButton>
          <MobileMenu.Item title="Home" url="/" />
          <MobileMenu.Item title="About Us" url="/" />
        </MobileMenu>
      </MobileMenuProvider>
    </SwipeProvider>
    );
  };
  `}
/>
