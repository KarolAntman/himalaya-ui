# useClipboard

Copy string to clipboard.

This is custom React hooks, you need to follow the <Link target="_blank" color href="https://reactjs.org/docs/hooks-rules.html">Basic Rules</Link> when you use it.

<Playground
  title="Sidebar Context Usage"
  desc="Access the sidebar context and interact with the sidebar's state."
  scope={{ useSidebar , Button}}
  code={`
() => {
  const { isEnabled, setIsEnabled } = useSidebar();
  const enableSidebar = () => {
    setIsEnabled(true);
  };
  const disableSidebar = () => {
    setIsEnabled(false);
  };
  return (
    <>
      <p>Sidebar is {isEnabled ? 'enabled' : 'disabled'}</p>
      <Button scale={0.5}  onClick={enableSidebar}>Enable Sidebar</Button>
      <Button ml="10px" scale={0.5}  onClick={disableSidebar}>Disable Sidebar</Button>
    </>
  );
}
`}
/>
